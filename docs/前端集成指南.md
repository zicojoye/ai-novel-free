## 📡 WebSocket连接流程

### 1. 客户端连接
```typescript
import { useWebSocket } from '@/hooks/useWebSocket'

function MyComponent() {
  const { isConnected, sendMessage } = useWebSocket({
    projectId: 1,
    roomId: 'workspace',
    onMessage: (message) => console.log('收到消息:', message),
    onEvent: (event) => console.log('收到事件:', event)
  })
  
  return (
    <div>
      <div>连接状态: {isConnected ? '已连接' : '未连接'}</div>
      <button onClick={() => sendMessage({
        type: 'message',
        content: '测试消息'
      })}>
        发送消息
      </button>
    </div>
  )
}
```

### 2. 消息类型
```typescript
// 聊天消息
{
  type: 'message',
  room_id: 'workspace',
  sender_id: 1,
  sender_type: 'user',
  content: '你好',
  timestamp: '2024-01-15T09:00:00Z'
}

// 事件流
{
  type: 'event',
  event_type: 'task_start',
  source: { id: 1, type: 'agent' },
  description: '开始创作第3章',
  importance: 8
}

// 艾特通知
{
  type: 'mention',
  agent_id: 1,
  message_id: 123,
  room_id: 'workspace',
  content: '@Agent1 请帮我润色',
  timestamp: '2024-01-15T09:00:00Z'
}
```

## 🔔 WebSocket重连策略

```typescript
// useWebSocket hook已实现重连逻辑
// 以下是完整的重连策略

// 1. 自动重连（指数退避）
const reconnectIntervals = [1000, 2000, 5000, 10000, 30000]
let reconnectAttempts = 0

const reconnect = () => {
  if (reconnectAttempts < reconnectIntervals.length) {
    const interval = reconnectIntervals[reconnectAttempts]
    setTimeout(connect, interval)
    reconnectAttempts++
  } else {
    // 最大重连次数后停止
    console.error('Max reconnection attempts reached')
  }
}

// 2. 心跳检测
const heartbeatInterval = setInterval(() => {
  if (isConnected) {
    sendMessage({ type: 'ping' })
  }
}, 30000)

// 3. 心跳超时检测
let lastPongTime = Date.now()
const heartbeatTimeout = 5000 // 5秒

const checkHeartbeat = () => {
  if (Date.now() - lastPongTime > heartbeatTimeout) {
    console.warn('Heartbeat timeout, reconnecting...')
    disconnect()
    reconnect()
  }
}

setInterval(checkHeartbeat, 10000)
```
