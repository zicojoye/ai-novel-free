**影响**: 无法在功能模块中获取当前项目ID

**当前路由配置**:
```tsx

<Route path="/chapter" element={<ChapterEditor />} />
<Route path="/plot" element={<PlotManager />} />
<Route path="/knowledge" element={<KnowledgeBase />} />
```

**应该改为**:
```tsx
// ✅ 需要项目ID参数

<Route path="/plot/:projectId" element={<PlotManager />} />
```

## 🔧 修复方案详解

### 修复1: 前后端数据集成 (最严重)

以 `WorldBuilding` 模块为例:

```tsx
// 当前代码 (使用模拟数据)
export default function WorldBuilding() {
  return (
    <div className="...">
      <textarea placeholder="描述故事的核心概念..." />
      {/* 所有输入框都没有状态管理和API调用 */}
    </div>
  )
}

// 修复后 (连接API)
export default function WorldBuilding() {
  const { projectId } = useParams()
  const [loading, setLoading] = useState(true)
  const [data, setData] = useState<WorldBuilding | null>(null)
  
  // 加载数据
  useEffect(() => {
    loadWorldBuilding()
  }, [projectId])
  
  const loadWorldBuilding = async () => {
    try {
      const response = await worldbuildingService.getWorldBuilding(projectId!)
      if (response.data) {
        setData(response.data)
      }
    } finally {
      setLoading(false)
    }
  }
  
  // 保存数据
  const handleSave = async () => {
    if (!data) return
    await worldbuildingService.updateWorldBuilding(projectId!, data)
  }
  
  if (loading) return <LoadingSpinner />
  
  return (
    <div className="...">
      <textarea 
        value={data?.core_concept || ''}
        onChange={(e) => setData({...data, core_concept: e.target.value})}
      />
      {/* 所有输入框都连接到state */}
      <button onClick={handleSave}>保存</button>
    </div>
  )
}
```

**需要修复的文件**: 6个模块 × 每个约50-100行代码 = 300-600行
